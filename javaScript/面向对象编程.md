```javascript

	原型 prototype
	原型链 __proto__
*/

/*
	很重要
*/
1. 所有的函数都以一个属性 prototype 对象属性 默认开辟一个堆内存
```

## 继承的东西

```javascript
/**
 * 封装：低耦合高内聚
 * 多态：重载和重写
 * 重写：类的继承中重写父类的方法
 */

/**
 * 面向对象: 一种编程思想，JS本身就是基于面向对象的 有很多类 例如 promise
 * 来管理异步编程
 */

/**
 * 继承
 * 1. 原型继承 实质是改变原型链指向
 */

function A(x) {
  this.x = x;
}
A.prototype.getX = function () {
  console.log(this.x);
};

function B(y) {
  this.y = y;
}

B.prototype = new A(200); // 原型继承 改写了B的prototype同时B有一个__proto__ 指向了A的prototype 这样就继承了属性的继承
B.prototype.constuctor = B; // 重新赋值constructor属性
B.prototype.getY = function () {
  console.log(this.y);
};
let b1 = new B(100);
b1.getX();

```



```javascript
/**
 * call继承
 * 特点:
 *  1. 使用call改变了this的指向
 *  2. 父类函数普通执行
 *  3. 只能继承父类的私有属性或者方法（原型的方法没有）
 *  4. 父类私有变成子类私有的
 */

function A(x) {
  this.x = x;
}

function B(y) {
  A.call(this, 100); // call继承
  this.y = y;
}
B.prototype.getY = function () {
  console.log(this.y);
};

```



```javascript
/**
 * 继承组合继承
 */

function A(x) {
  this.x = x;
}

A.prototype.getX = function () {
  console.log(this.x);
};
// Object.create(OBJ) 创建一个空对象 让空对象的__proto__ 指向OBJ
B.prototype = Object.create(A.prototype); // {}.__proto__ = A.prototype
B.prototype.constructor = B;
function B(y) {
  A.call(this, 200);
  this.y = y;
}

B.prototype.getY = function () {
  console.log(this.y);
};

// Object.create() 实现
function create(obj) {
  function fn() {}
  Fn.prototype = obj;
  return new Fn();
}
```
